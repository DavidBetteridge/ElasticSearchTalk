Match Search
POST /meetups/_search
{
    "query" : {
        "match" : { "speakers": "Kevin"} 
    }
}

Match Search (with and)
POST /meetups/_search
{
  "query": {
    "match": {
      "speakers": {
"query" : "Kevin Betteridge",
"operator" : "and"
    }
    }
  }
}

Term Search (no results)
POST /meetups/_search
{
    "query" : {
        "term" : { "speakers": "Kevin Smith"} 
    }
}

Term Search (one result)
POST /meetups/_search
{ 
	"query" : {
        "term" : { "_id": "dec2017"} 
    }
}

Term Search (fixed)

{
    "query" : {
        "term" : { "speakers.keyword": "Kevin Smith"} 
    }
